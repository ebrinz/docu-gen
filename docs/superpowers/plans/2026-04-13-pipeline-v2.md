```python
"""Timing model: compute per-clip duration and chapter offsets from WAV + word_times.

All timing derives from actual audio data. No hardcoded buffers.
"""

from pathlib import Path
from scipy.io import wavfile

PACING_TARGETS = {"tight": 0.5, "normal": 1.5, "breathe": 3.0}
FIXED_CARD_DURATION = 3.0


def get_wav_duration(wav_path: str | Path) -> float:
    """Measure WAV file duration in seconds."""
    sr, data = wavfile.read(str(wav_path))
    if data.ndim > 1:
        data = data[:, 0]
    return len(data) / sr


def compute_clip_timing(clip: dict, wav_duration: float,
                        chapter_offset: float) -> dict:
    """Compute timing fields for a single clip.

    Args:
        clip: Clip dict with word_times and pacing fields.
        wav_duration: Measured WAV duration (0.0 if no narration).
        chapter_offset: Cumulative time from chapter start.

    Returns:
        Timing dict with wav_duration, speech_start, speech_end,
        trailing_silence, pacing_pad, clip_duration, chapter_offset.
    """
    word_times = clip.get("word_times", [])
    pacing = clip.get("pacing", "normal")

    # No narration — fixed duration chapter card
    if not word_times or wav_duration <= 0:
        return {
            "wav_duration": 0.0,
            "speech_start": 0.0,
            "speech_end": 0.0,
            "trailing_silence": 0.0,
            "pacing_pad": 0.0,
            "clip_duration": FIXED_CARD_DURATION,
            "chapter_offset": chapter_offset,
        }

    speech_start = word_times[0].get("start", 0.0)
    speech_end = word_times[-1].get("end", wav_duration)
    trailing_silence = max(0.0, wav_duration - speech_end)

    target_pad = PACING_TARGETS.get(pacing, 1.5)
    pacing_pad = max(0.0, target_pad - trailing_silence)

    clip_duration = wav_duration + pacing_pad

    return {
        "wav_duration": wav_duration,
        "speech_start": speech_start,
        "speech_end": speech_end,
        "trailing_silence": round(trailing_silence, 3),
        "pacing_pad": round(pacing_pad, 3),
        "clip_duration": round(clip_duration, 3),
        "chapter_offset": round(chapter_offset, 3),
    }


def compute_chapter_timeline(clips: list[dict]) -> list[float]:
    """Compute chapter_offset for each clip from its timing.clip_duration.

    Args:
        clips: List of clip dicts, each with timing.clip_duration set.

    Returns:
        List of chapter_offset floats, one per clip.
    """
    offsets = []
    offset = 0.0
    for clip in clips:
        offsets.append(round(offset, 3))
        offset += clip.get("timing", {}).get("clip_duration", 3.0)
    return offsets
```

"""Slide type registry — defines available scene builders and their valid cue events.

The direct tool picks slide types from this registry. The renderer dispatches to
scene builders based on slide_type. This module is the contract between them.
"""

SLIDE_REGISTRY = {
    "title": {
        "description": "Title + subtitle with configurable reveal style",
        "events": {"reveal_title", "reveal_subtitle"},
        "params": {"reveal_style": "particle"},
    },
    "chapter_card": {
        "description": "Chapter number + title, imperial border draws in",
        "events": {"reveal_number", "reveal_title"},
        "params": {},
    },
    "svg_reveal": {
        "description": "SVG asset fades/draws in, Ken Burns drift, labels",
        "events": {"show_asset", "highlight_region", "show_label"},
        "params": {},
    },
    "photo_organism": {
        "description": "Photo inset with HUD border, animated pointer labels",
        "events": {"show_photo", "show_structure", "show_name", "show_note"},
        "params": {},
    },
    "counter_sync": {
        "description": "Number animates 0 to target, keyed to narration word",
        "events": {"start_count", "hold"},
        "params": {"to": 0, "color": "gold", "label": ""},
    },
    "bar_chart_build": {
        "description": "Horizontal bars grow in sequence, synced to list",
        "events": {"show_bar"},
        "params": {"items": []},
    },

        "description": "Side-by-side comparison with morphing values",
        "events": {"show_before", "show_after", "morph"},
        "params": {"label": "", "before": "", "after": ""},
    },
    "dot_merge": {
        "description": "Two compound dots approach and merge",
        "events": {"show_dot1", "show_dot2", "merge"},
        "params": {"dot1": "", "dot2": "", "result": ""},
    },
    "remove_reveal": {
        "description": "One compound fades, another emerges",
        "events": {"remove", "reveal"},
        "params": {"removed": "", "emerged": ""},
    },
    "data_text": {
        "description": "Key text/numbers on screen synced to narration",
        "events": {"show_text"},
        "params": {"text": ""},
    },
    "ambient_field": {
        "description": "Particle field + theme bg, no foreground content",
        "events": set(),
        "params": {},
    },
}


def validate_slide_type(slide_type: str) -> bool:
    """Check if a slide type exists in the registry."""
    return slide_type in SLIDE_REGISTRY


def validate_cue_event(slide_type: str, event: str) -> bool:
    """Check if an event is valid for a given slide type."""
    info = SLIDE_REGISTRY.get(slide_type)
    if not info:
        return False
    return event in info["events"]


def get_slide_types_prompt() -> str:
    """Format the registry as a string for inclusion in LLM prompts."""
    lines = []
    for name, info in SLIDE_REGISTRY.items():
        events = ", ".join(sorted(info["events"])) if info["events"] else "(none)"
        lines.append(f"- {name}: {info['description']}. Cue events: {events}")
    return "\n".join(lines)
```

VALID_TRANSITIONS = {"crossfade", "cut", "wipe_left", "fade_black", "crossfade_next"}
THEME_SOUNDS = {
    "imperial_chime", "dark_swell", "crystal_ping", "heartbeat",
    "bell", "sonar", "resolve", "tension",
}


def validate_clip_direction(direction: dict, clip: dict,
                            available_assets: set[str]) -> list[str]:
    """Validate a single clip's visual direction. Returns list of error strings."""
    errors = []
    clip_id = clip.get("clip_id", "unknown")

    # Check required fields
    for field in REQUIRED_FIELDS:
        if field not in direction:
            errors.append(f"{clip_id}: missing required field '{field}'")

    # Validate slide_type
    slide_type = direction.get("slide_type", "")
    if slide_type and not validate_slide_type(slide_type):
        errors.append(f"{clip_id}: invalid slide_type '{slide_type}'")

    # Validate assets exist
    for asset in direction.get("assets", []):
        if asset not in available_assets:
            errors.append(f"{clip_id}: asset not found '{asset}'")

    # Validate cue_words
    word_times = clip.get("word_times", [])
    for cue in direction.get("cue_words", []):
        idx = cue.get("at_index", -1)
        if idx < 0 or idx >= len(word_times):
            errors.append(f"{clip_id}: cue_words at_index {idx} out of bounds (max {len(word_times) - 1})")
        event = cue.get("event", "")
        if slide_type and event and not validate_cue_event(slide_type, event):
            errors.append(f"{clip_id}: invalid event '{event}' for slide_type '{slide_type}'")

    # Validate layout
    layout = direction.get("layout", "")
    if layout and layout not in VALID_LAYOUTS:
        errors.append(f"{clip_id}: invalid layout '{layout}'")

    # Validate transitions
    for field in ("transition_in", "transition_out"):
        val = direction.get(field, "")
        if val and val not in VALID_TRANSITIONS:
            errors.append(f"{clip_id}: invalid {field} '{val}'")

    # Validate transition_sound
    sound = direction.get("transition_sound")
    if sound is not None and sound not in THEME_SOUNDS:
        errors.append(f"{clip_id}: invalid transition_sound '{sound}'")

    return errors


def validate_all_clips(clips_data: dict, available_assets: set[str]) -> list[str]:
    """Validate all clips in clips_data. Returns aggregated error list."""
    all_errors = []
    for chapter in clips_data.get("chapters", []):
        for clip in chapter.get("clips", []):
            direction = clip.get("visuals", {})
            errors = validate_clip_direction(direction, clip, available_assets)
            all_errors.extend(errors)
    return all_errors


def _build_prompt(production_plan: dict, clips_data: dict,
                  available_assets: list[str], creative_notes: str) -> str:
    """Build the Claude prompt for creative direction."""
    slide_types_desc = get_slide_types_prompt()
    assets_list = "\n".join(f"  - {a}" for a in sorted(available_assets))

    # Build clip summaries with narration + word count
    clip_summaries = []
    for chapter in clips_data["chapters"]:
        ch_id = chapter["id"]
        ch_title = chapter.get("title", "")
        plan_chapter = next(
            (c for c in production_plan.get("chapters", []) if c["id"] == ch_id), {}
        )
        plan_visuals = plan_chapter.get("visuals", {}).get("manim", "")
        plan_assets = (
            plan_chapter.get("visuals", {}).get("existing_svg", [])
            + plan_chapter.get("visuals", {}).get("source_images", [])
            + [s.get("id", "") for s in plan_chapter.get("visuals", {}).get("new_svg", [])]
        )

        clip_summaries.append(f"\n## Chapter: {ch_id} — {ch_title}")
        clip_summaries.append(f"Visual direction from plan: {plan_visuals}")
        clip_summaries.append(f"Planned assets: {plan_assets}")

        for clip in chapter["clips"]:
            cid = clip["clip_id"]
            text = clip.get("text", "(no narration)")
            n_words = len(clip.get("word_times", []))
            clip_summaries.append(f"\n  Clip {cid} ({n_words} words): {text}")

    clips_block = "\n".join(clip_summaries)

    return f"""You are a creative director for a documentary video. Assign visual direction to every clip.

AVAILABLE SLIDE TYPES:
{slide_types_desc}

AVAILABLE IMAGE ASSETS:
{assets_list}

PRODUCTION PLAN CLIPS:
{clips_block}

{f"CREATIVE NOTES: {creative_notes}" if creative_notes else ""}

For EVERY clip listed above, output a JSON object with these REQUIRED fields:
- "slide_type": one of the slide types listed above
- "assets": list of filenames from the available assets (empty list if none)
- "cue_words": list of {{"word": "...", "at_index": N, "event": "...", "params": {{}}}}
  - at_index is the 0-based index into the clip's word_times array
  - event must be valid for the chosen slide_type
  - params are slide-type-specific parameters
- "layout": one of "center", "split_left_right", "bottom_third", "full_bleed"
- "transition_in": one of "crossfade", "cut", "wipe_left", "fade_black"
- "transition_out": one of "crossfade_next", "cut", "fade_black"
- "transition_sound": one of "imperial_chime", "dark_swell", "crystal_ping", "heartbeat", "bell", "sonar", "resolve", "tension", or null

RULES:
1. Match assets to narration content — show the survival curve SVG when survival is discussed, show the sponge photo when sponges are narrated.
2. Use cue_words to sync visuals to specific words. A counter should start when the number is spoken. A photo should appear when the organism is named.
3. Chapter cards (clips with no narration text) should use slide_type "chapter_card".
4. The first clip of the entire video (intro_01) should use slide_type "title" with params.reveal_style = "particle".
5. Use transition_sound only at chapter boundaries, not between every clip.
6. Prefer variety — don't assign the same slide_type to 5 consecutive clips.

Output format: a JSON object where keys are clip_ids and values are the direction objects.
Example:
```json
{{
  "intro_01": {{
    "slide_type": "title",
    "assets": [],
    "cue_words": [],
    "layout": "center",
    "transition_in": "fade_black",
    "transition_out": "crossfade_next",
    "transition_sound": null,
    "params": {{"reveal_style": "particle"}}
  }},
  "intro_02": {{
    "slide_type": "counter_sync",
    "assets": [],
    "cue_words": [{{"word": "billion", "at_index": 8, "event": "start_count", "params": {{"to": 420, "color": "gold"}}}}],
    "layout": "center",
    "transition_in": "crossfade",
    "transition_out": "crossfade_next",
    "transition_sound": null
  }}
}}
```

Respond with ONLY the JSON object. No markdown fences, no explanation."""


def _parse_llm_response(response_text: str) -> dict:
    """Parse the LLM response JSON, handling markdown fences if present."""
    text = response_text.strip()
    if text.startswith("```"):
        # Strip markdown code fences
        lines = text.split("\n")
        lines = [l for l in lines if not l.strip().startswith("```")]
        text = "\n".join(lines)
    return json.loads(text)


def direct_clips(project_path: str | Path, creative_notes: str = "") -> str:
    """Assign visual direction to every clip using Claude.

    Reads production plan, clips.json (with word_times), and available assets.
    Makes one Claude API call. Validates output. Computes timing. Writes back
    to clips.json.

    Returns summary string.
    """
    project_path = Path(project_path)
    config = load_config(project_path)
    build_dir = project_path / "build"
    images_dir = project_path / "images"
    narr_dir = build_dir / "narration"

    # Load inputs
    clips_data = json.loads((build_dir / "clips.json").read_text())
    plan_path = build_dir / "plan.json"
    production_plan = json.loads(plan_path.read_text()) if plan_path.exists() else {}

    available_assets = {f.name for f in images_dir.iterdir()} if images_dir.exists() else set()

    # Build prompt and call Claude
    prompt = _build_prompt(production_plan, clips_data, sorted(available_assets), creative_notes)

    client = Anthropic()
    message = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=8000,
        messages=[{"role": "user", "content": prompt}],
    )

    response_text = message.content[0].text
    directions = _parse_llm_response(response_text)

    # Apply directions to clips
    applied = 0
    for chapter in clips_data["chapters"]:
        for clip in chapter["clips"]:
            clip_id = clip["clip_id"]
            if clip_id in directions:
                clip["visuals"] = directions[clip_id]
                applied += 1

    # Validate
    errors = validate_all_clips(clips_data, available_assets)
    if errors:
        # Write what we have so user can fix
        (build_dir / "clips.json").write_text(json.dumps(clips_data, indent=2) + "\n")
        error_list = "\n".join(f"  - {e}" for e in errors)
        return (
            f"Applied direction to {applied} clips but validation found {len(errors)} errors:\n"
            f"{error_list}\n\n"
            f"clips.json updated — fix errors and re-run direct, or proceed to render."
        )

    # Compute timing for all clips
    for chapter in clips_data["chapters"]:
        offset = 0.0
        for clip in chapter["clips"]:
            clip_id = clip["clip_id"]
            wav_path = narr_dir / f"{clip_id}.wav"
            wav_dur = get_wav_duration(wav_path) if wav_path.exists() else 0.0
            clip["timing"] = compute_clip_timing(clip, wav_dur, offset)
            offset += clip["timing"]["clip_duration"]

    # Write back
    (build_dir / "clips.json").write_text(json.dumps(clips_data, indent=2) + "\n")

    return f"Directed {applied} clips. Timing computed. All validation passed. clips.json updated."
```

Reveal styles:
- particle: Dots converge into title letterforms, scanline subtitle
- glitch: RGB offset + scanlines, locks in over 1-2s
- trace: Stroke paths draw then fill with gradient sweep

def build_title_script(title: str, subtitle: str, reveal_style: str,
                       duration: float, colors: dict, font_dir: str) -> str:
    """Generate a Manim Python script for a title card.

    Args:
        title: Main title text.
        subtitle: Subtitle text.
        reveal_style: One of particle, glitch, trace, typewriter.
        duration: Total scene duration in seconds.
        colors: Dict with bg, accent_gold, accent_cyan, text keys.
        font_dir: Absolute path to directory containing Rajdhani TTFs.

    Returns:
        Complete Manim Python script as string.
    """
    bg = colors.get("bg", "#0a0e27")
    gold = colors.get("accent_gold", "#f59e0b")
    cyan = colors.get("accent_cyan", "#22d3ee")
    text_color = colors.get("text", "#e2e8f0")
    title_esc = title.replace('"', '\\"')
    subtitle_esc = subtitle.replace('"', '\\"')
    font_dir_esc = font_dir.replace("\\", "\\\\")

    builders = {
        "particle": _particle_script,
        "glitch": _glitch_script,
        "trace": _trace_script,
        "typewriter": _typewriter_script,
    }
    builder = builders.get(reveal_style, _particle_script)
    return builder(title_esc, subtitle_esc, duration, bg, gold, cyan, text_color, font_dir_esc)


def _particle_script(title, subtitle, duration, bg, gold, cyan, text_color, font_dir):
    return f'''from manim import *
import numpy as np

config.background_color = "{bg}"

class Scene_title(Scene):
    def construct(self):
        # Register Rajdhani font
        import manimpango
        manimpango.register_font("{font_dir}/Rajdhani-Bold.ttf")

        # Create title text to get target positions
        title_text = Text("{title}", font="Rajdhani", weight=BOLD,
                         color="{gold}").scale(2.0)
        subtitle_text = Text("{subtitle}", font="Rajdhani",
                            color="{cyan}").scale(0.55)
        subtitle_text.next_to(title_text, DOWN, buff=0.5)

        # Create particle field — 300 dots drifting
        np.random.seed(42)
        n_particles = 300
        dots = VGroup()
        start_positions = []
        for _ in range(n_particles):
            x = np.random.uniform(-8, 8)
            y = np.random.uniform(-5, 5)
            start_positions.append(np.array([x, y, 0]))
            dot = Dot(point=[x, y, 0], radius=0.03,
                     color="{gold}", fill_opacity=0.4)
            dots.add(dot)

        self.add(dots)

        # Drift phase — particles float gently
        drift_anims = []
        for i, dot in enumerate(dots):
            dx = np.random.uniform(-0.5, 0.5)
            dy = np.random.uniform(-0.3, 0.3)
            target = dot.get_center() + np.array([dx, dy, 0])
            drift_anims.append(dot.animate.move_to(target))
        self.play(*drift_anims, run_time=2.0, rate_func=smooth)

        # Convergence phase — particles rush to title letter positions
        # Sample target points along the title text boundary
        title_text_temp = title_text.copy()
        self.add(title_text_temp)
        title_text_temp.set_opacity(0)

        # Get points from the title text paths
        target_points = []
        for submob in title_text_temp:
            points = submob.get_all_points()
            if len(points) > 0:
                indices = np.linspace(0, len(points) - 1,
                                     min(len(points), n_particles // len(title_text_temp) + 1),
                                     dtype=int)
                for idx in indices:
                    target_points.append(points[idx])

        # Pad or trim target_points to match n_particles
        while len(target_points) < n_particles:
            target_points.append(target_points[len(target_points) % max(len(target_points), 1)])
        target_points = target_points[:n_particles]
        np.random.shuffle(target_points)

        converge_anims = []
        for dot, target in zip(dots, target_points):
            converge_anims.append(dot.animate.move_to(target).set_opacity(0.8))
        self.play(*converge_anims, run_time=2.5, rate_func=rush_into)

        # Flash and reveal title
        flash = Rectangle(width=16, height=10, fill_color="{gold}",
                         fill_opacity=0.3, stroke_width=0)
        self.play(FadeIn(flash, run_time=0.15))
        self.play(
            FadeOut(flash, run_time=0.3),
            FadeOut(dots, run_time=0.3),
            FadeIn(title_text, run_time=0.3),
        )

        # Scanline subtitle reveal
        scan = Line(LEFT * 7, RIGHT * 7, color="{cyan}", stroke_width=1.5)
        scan.move_to(subtitle_text.get_top() + UP * 0.2)
        self.play(
            scan.animate.move_to(subtitle_text.get_bottom() + DOWN * 0.2),
            FadeIn(subtitle_text),
            run_time=1.2,
            rate_func=linear,
        )
        self.play(FadeOut(scan), run_time=0.3)

        # Hold
        hold_time = max({duration} - 6.5, 1.0)
        self.wait(hold_time)

        # Fade out
        self.play(FadeOut(title_text), FadeOut(subtitle_text), run_time=1.5)
'''

            word_count = len(clip.get("text", "").split())
        exagg = clip.get("exaggeration", 0.0)
        if word_count <= SHORT_WORD_LIMIT and exagg >= HIGH_EXAGGERATION:
            indices.append(i)
    return indices


def _plan_consolidation(clips: list[dict],
                        short_indices: list[int]) -> list[dict]:
    """Plan which short clips merge into which neighbors.

    Prefers merging into the preceding clip (setup → punchline).
    Returns list of {"merge_into": int, "short_index": int}.
    """
    plan = []
    for idx in short_indices:
        if idx > 0:
            plan.append({"merge_into": idx - 1, "short_index": idx})
        elif idx < len(clips) - 1:
            plan.append({"merge_into": idx + 1, "short_index": idx})
        # If it's the only clip, skip — can't merge
    return plan
